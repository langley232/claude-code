## 🛠 **Implementation Code Examples**

### **Instagram Profile & Posts Search**
```javascript
class InstagramSearchTool {
    constructor(mcpTools) {
        this.mcp = mcpTools;
    }

    async searchProfile(username) {
        try {
            const profileData = await this.mcp.call_tool(
                "web_data_instagram_profiles",
                { url: `https://instagram.com/${username}` }
            );
            
            return {
                platform: 'instagram',
                type: 'profile',
                data: {
                    username: profileData.username,
                    fullName: profileData.full_name,
                    bio: profileData.biography,
                    followers: profileData.follower_count,
                    following: profileData.following_count,
                    posts: profileData.post_count,
                    verified: profileData.is_verified,
                    profilePic: profileData.profile_picture_url
                }
            };
        } catch (error) {
            console.error('Instagram profile search error:', error);
            return null;
        }
    }

    async searchPosts(username, limit = 10) {
        try {
            const postsData = await this.mcp.call_tool(
                "web_data_instagram_posts",
                { 
                    url: `https://instagram.com/${username}`,
                    limit: limit 
                }
            );
            
            return postsData.map(post => ({
                platform: 'instagram',
                type: 'post',
                id: post.shortcode,
                content: post.caption,
                mediaUrl: post.display_url,
                likes: post.like_count,
                comments: post.comment_count,
                timestamp: post.taken_at_timestamp,
                hashtags: this.extractHashtags(post.caption)
            }));
        } catch (error) {
            console.error('Instagram posts search error:', error);
            return [];
        }
    }
}
```

### **Multi-Platform Search Manager**
```javascript
class UniversalSocialMediaSearch {
    constructor(mcpTools) {
        this.instagram = new InstagramSearchTool(mcpTools);
        this.tiktok = new TikTokSearchTool(mcpTools);
        this.linkedin = new LinkedInSearchTool(mcpTools);
        this.facebook = new FacebookSearchTool(mcpTools);
        this.twitter = new TwitterSearchTool(mcpTools);
        this.youtube = new YouTubeSearchTool(mcpTools);
        this.reddit = new RedditSearchTool(mcpTools);
    }

    async searchAllPlatforms(query, platforms = ['all']) {
        const results = {
            instagram: [],
            tiktok: [],
            linkedin: [],
            facebook: [],
            twitter: [],
            youtube: [],
            reddit: []
        };

        const searchPromises = [];

        if (platforms.includes('all') || platforms.includes('instagram')) {
            searchPromises.push(
                this.instagram.searchPosts(query).then(data => {
                    results.instagram = data;
                })
            );
        }

        if (platforms.includes('all') || platforms.includes('tiktok')) {
            searchPromises.push(
                this.tiktok.searchVideos(query).then(data => {
                    results.tiktok = data;
                })
            );
        }

        if (platforms.includes('all') || platforms.includes('linkedin')) {
            searchPromises.push(
                this.linkedin.searchPeople(query).then(data => {
                    results.linkedin = data;
                })
            );
        }

        // Add other platforms...

        await Promise.allSettled(searchPromises);
        
        return this.normalizeResults(results);
    }

    normalizeResults(results) {
        const normalized = [];
        
        Object.entries(results).forEach(([platform, data]) => {
            data.forEach(item => {
                normalized.push({
                    ...item,
                    platform: platform,
                    searchTimestamp: new Date().toISOString()
                });
            });
        });
        
        // Sort by engagement, recency, or relevance
        return normalized.sort((a, b) => {
            return this.calculateRelevanceScore(b) - this.calculateRelevanceScore(a);
        });
    }

    calculateRelevanceScore(item) {
        let score = 0;
        
        // Engagement score
        const engagement = (item.likes || 0) + (item.comments || 0) + (item.shares || 0);
        score += Math.log(engagement + 1) * 10;
        
        // Recency score
        const ageInHours = (Date.now() - new Date(item.timestamp)) / (1000 * 60 * 60);
        score += Math.max(0, 100 - ageInHours);
        
        // Platform weighting
        const platformWeights = {
            instagram: 1.2,
            tiktok: 1.3,
            linkedin: 1.1,
            twitter: 1.0,
            youtube: 1.1,
            facebook: 0.9,
            reddit: 0.8
        };
        score *= platformWeights[item.platform] || 1.0;
        
        return score;
    }
}
```
